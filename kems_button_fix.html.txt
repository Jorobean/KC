<script>
const button = document.getElementById("button");
const log = document.getElementById("log");
const pill = document.getElementById("pill");

// Simulated sounds (since audio files don't exist)
const sounds = {
  boot: { play: () => {} },
  recordStart: { play: () => {} },
  recordStop: { play: () => {} },
  shutdown: { play: () => {} }
};

let pressTimer;
let pressCount = 0;
let lastPressTime = 0;
let recording = false;
let deviceOn = false;
let holdTimeout;
let multiPressTimeout;
let isTyping = false;
let firstInteraction = true;

function typeText(text, callback) {
  isTyping = true;
  let i = 0;
  const div = document.createElement('div');
  log.appendChild(div);
  
  function type() {
    if (i < text.length) {
      div.textContent += text.charAt(i);
      i++;
      log.scrollTop = log.scrollHeight;
      setTimeout(type, 30); // 30ms per character for typing effect
    } else {
      log.scrollTop = log.scrollHeight;
      isTyping = false;
      if (callback) callback();
    }
  }
  type();
}

function logAction(msg) {
  if (firstInteraction) {
    log.innerHTML = ''; // Clear initial text on first interaction
    firstInteraction = false;
  }
  typeText(msg);
}

function updatePill(state) {
  pill.className = "";
  pill.style.cssText = ""; // Clear all inline styles
  
  if (!deviceOn) {
    pill.style.background = "transparent";
    pill.style.opacity = "0.3";
    pill.style.border = "2px solid #444";
  } else if (state === "boot") {
    pill.classList.add("booting");
  } else if (state === "recording") {
    pill.classList.add("recording");
  } else if (state === "on") {
    pill.classList.add("on");
  }
}

function handlePress() {
  // Device is OFF - single press boots it
  if (!deviceOn && pressCount === 1) {
    sounds.boot.play();
    logAction("Booting device...");
    deviceOn = true;
    updatePill("boot");
    setTimeout(() => { 
      updatePill("on"); 
      pressCount = 0;
    }, 1000);
    return;
  }
  
  // Device is ON, not recording - single press starts recording
  if (deviceOn && !recording && pressCount === 1) {
    sounds.recordStart.play();
    logAction("Recording...");
    recording = true;
    updatePill("recording");
    pressCount = 0;
    return;
  }
  
  // Device is ON, recording - single press stops recording
  if (deviceOn && recording && pressCount === 1) {
    sounds.recordStop.play();
    logAction("Sending recording...");
    recording = false;
    updatePill("on");
    pressCount = 0;
    return;
  }
}

// Mouse events for desktop testing
button.addEventListener("mousedown", (e) => {
  pressTimer = Date.now();
  
  holdTimeout = setTimeout(() => {
    if (deviceOn) {
      sounds.shutdown.play();
      logAction("Shutting down...");
      recording = false;
      deviceOn = false;
      updatePill("off");
      pressCount = 0;
      clearTimeout(multiPressTimeout);
    }
  }, 3000);
});

button.addEventListener("mouseup", (e) => {
  clearTimeout(holdTimeout);
  
  const holdTime = (Date.now() - pressTimer) / 1000;
  
  if (holdTime >= 3) {
    return; // Shutdown already handled
  }
  
  const now = Date.now();
  if (now - lastPressTime > 1000) {
    pressCount = 0;
  }
  
  pressCount++;
  lastPressTime = now;
  
  // Clear existing timeout
  clearTimeout(multiPressTimeout);
  
  // Wait briefly to see if more presses are coming
  multiPressTimeout = setTimeout(() => {
    handlePress();
  }, 300);
});

// Touch events for mobile
button.addEventListener("touchstart", (e) => {
  e.preventDefault();
  pressTimer = Date.now();
  
  holdTimeout = setTimeout(() => {
    if (deviceOn) {
      sounds.shutdown.play();
      logAction("Shutting down...");
      recording = false;
      deviceOn = false;
      updatePill("off");
      pressCount = 0;
      clearTimeout(multiPressTimeout);
    }
  }, 3000);
});

button.addEventListener("touchend", (e) => {
  e.preventDefault();
  clearTimeout(holdTimeout);
  
  const holdTime = (Date.now() - pressTimer) / 1000;
  
  if (holdTime >= 3) {
    return;
  }
  
  const now = Date.now();
  if (now - lastPressTime > 1000) {
    pressCount = 0;
  }
  
  pressCount++;
  lastPressTime = now;
  
  clearTimeout(multiPressTimeout);
  
  multiPressTimeout = setTimeout(() => {
    handlePress();
  }, 300);
});

typeText("KEMS Button Simulator");
typeText("Press once to boot, once to record, once to stop");
typeText("Hold 3s to shutdown");
</script>
